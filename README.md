# Medical-Chatbot

# 1. Clone the repository

git clone https://github.com/miteshupadhyay/Medical-Chatbot.git

# 2. Create virtual environment

conda create -p medibot python==3.11 -y

# 3. Activate the environment

conda activate D:\GenAI\Medical-Chatbot\medibot

# 4. Install Dependencies

pip install -r requirements.txt


python store_index.py

python app.py
